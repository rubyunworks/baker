    def load
      if FileTest::exists?(SETTINGS.cat_config_file)
        settings = YAML::load(File.new(SETTINGS.cat_config_file))
      else
        raise BakerError, "could not find catalog settings file at:\n#{SETTINGS.cat_config_file}"
      end
      @package_dir = settings['source_dir']
      @catalog_dir = settings['catalog_dir']
      @personal_dir = settings['personal_dir']
      @local_region = settings['local_region']
      @catalog_urls = settings['catalog_urls']
    end
