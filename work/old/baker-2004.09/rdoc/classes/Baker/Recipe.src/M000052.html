    def compliment_and_supplement
      comp = @compliment.find_all do |c|
        !@@buildconfig.avoid.collect{ |a| a.downcase }.include?(c.split(/(\s+|--)/)[0].downcase)
      end
      sup = @supplement.find_all do |s|
        @@buildconfig.ensure.collect{ |e| e.downcase }.include?(s.split(/(\s+|--)/)[0].downcase)
      end
      return (comp + sup).uniq
    end
