    def installed?
      success = false
      @install_indicators.each do |ii|
        if ii.include?(' ')
          root, pattern, exclusions = ii.split(' ')
          success = findfile(root, pattern, exclusions) ? true : false
        else
          success = FileTest::exists?(ii)
        end
        break if success
      end
      success
    end
