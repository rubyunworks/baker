    def resolve_dependencies #(catalog)
      @resolved_dependencies = [] unless @resolved_dependencies
      @unresolved_dependencies = [] unless @unresolved_dependencies
      @dependencies.concat(self.compliment_and_supplement)
      @dependencies.each do |unresolved_name|
        recipe_name = resolve_name(unresolved_name)
        if recipe_name
          @resolved_dependencies << recipe_name  # :)
        else
          @unresolved_dependencies << unresolved_name  # :(
        end
      end
    end
