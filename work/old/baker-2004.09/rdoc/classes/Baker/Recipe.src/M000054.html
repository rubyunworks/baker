    def package_file
      pkg_file = nil
      @source_urls.each do |url|
        poss_pkg_file = File.join(self.package_dir, File.basename(url[0]))
        if File::exists?(poss_pkg_file)
          pkg_file = poss_pkg_file
          break
        end
      end
      pkg_file
    end
