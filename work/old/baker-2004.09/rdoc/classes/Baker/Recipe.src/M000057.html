    def fetch
      # source directory already exists? if not then create
      pkg_dir = self.package_dir
      if !File.directory?(pkg_dir)
        File.makedirs(pkg_dir)
      end
      # have download manger do a region based download
      success = DownloadManager.regional_download(self.source_urls,@@catalog.local_region,pkg_dir,SETTINGS.opt_force)
      if !success
        raise BakerError, "source package fetch failed"
      end
    end
