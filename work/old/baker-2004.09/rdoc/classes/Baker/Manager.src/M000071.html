    def missing
      missing_dependencies = {}
      @prioritized_recipes.each do |recipe_name|
        if @recipe_lookup[recipe_name].unresolved_dependencies.length > 0
          missing_dependencies[recipe_name] = @recipe_lookup[recipe_name].unresolved_dependencies
        end
      end
      return missing_dependencies
    end
